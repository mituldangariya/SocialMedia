CREATE PROCEDURE [dbo].[AddFriendProcedure]
    @UserId INT,
    @FollowerId INT
AS
BEGIN
    DECLARE @CurrentUserId INT;
    SET @CurrentUserId = @FollowerId;

    -- Check if the user is trying to add themselves as a friend
    IF @UserId = @CurrentUserId
    BEGIN
        RAISERROR ('You cannot add yourself as a friend.', 16, 1);
        RETURN;
    END;

    -- Check if friendship already exists
    IF EXISTS (
        SELECT 1
        FROM UserFriends
        WHERE (UserId = @CurrentUserId AND FollowerId = @UserId)
            OR (UserId = @UserId AND FollowerId = @CurrentUserId)
    )
    BEGIN
        RAISERROR ('Friendship already exists.', 16, 1);
        RETURN;
    END;

    -- Add friendship
    INSERT INTO UserFriends (UserId, FollowerId, IsFriend, RequestStatus)
    VALUES (@CurrentUserId, @UserId, 1, 'pending');

    RETURN;
END;
GO

CREATE PROCEDURE [dbo].[CheckEmail]
    @UserId INT,
    @Email NVARCHAR(255)
AS
BEGIN
    SET NOCOUNT ON;

    IF @Email IS NULL OR LTRIM(RTRIM(@Email)) = ''
    BEGIN
        -- Invalid email
        SELECT 'Invalid email' AS ResultMessage;
        RETURN;
    END

    DECLARE @ExistingUserId INT;
    SELECT @ExistingUserId = UserId
    FROM UserData
    WHERE Email = @Email;

    IF (@ExistingUserId IS NULL)
    BEGIN
        -- Email not in use
        SELECT 0 AS EmailInUse;
    END
    ELSE IF (@ExistingUserId = @UserId)
    BEGIN
        -- Email belongs to the same user
        SELECT 0 AS EmailInUse;
    END
    ELSE
    BEGIN
        -- Email already in use by another user
        SELECT 1 AS EmailInUse;
    END;
END
GO

CREATE PROCEDURE [dbo].[DeleteComment]
    @commentId INT
AS
BEGIN
    SET NOCOUNT ON;

    -- Declare variables
    DECLARE @postId INT;
    DECLARE @deletedCommentsCount INT;

    BEGIN TRY
        -- Begin a transaction
        BEGIN TRANSACTION;

        -- Retrieve the postId for the given commentId
        SELECT @postId = PostId
        FROM PostComments
        WHERE CommentId = @commentId;

        -- Check if the comment exists
        IF (@postId IS NOT NULL)
        BEGIN
            -- Count the number of comments to be marked as deleted
            SET @deletedCommentsCount = (
                SELECT COUNT(*)
                FROM PostComments
                WHERE CommentId = @commentId
                   OR ParentCommentId = @commentId
                   AND IsDeleted = 0
            );

            -- Mark the main comment and its child comments as deleted
            UPDATE PostComments
            SET IsDeleted = 1
            WHERE CommentId = @commentId
               OR ParentCommentId = @commentId;

            -- Decrease the comment count for the post by the number of newly deleted comments
            UPDATE UserPosts
            SET CommentCount = CommentCount - @deletedCommentsCount
            WHERE PostId = @postId;

            -- Commit the transaction
            COMMIT TRANSACTION;
        END
        ELSE
        BEGIN
            -- Rollback the transaction if the comment does not exist
            ROLLBACK TRANSACTION;
            RAISERROR('Comment does not exist or has already been deleted.', 16, 1);
        END
    END TRY
    BEGIN CATCH
        -- Rollback the transaction in case of error
        IF @@TRANCOUNT > 0
            ROLLBACK TRANSACTION;

        -- Re-throw the error
        THROW;
    END CATCH;
END
GO

CREATE PROCEDURE [dbo].[GetUserData]
    @CurrentUserId INT
AS
BEGIN
    SELECT 
        u.UserId,
        u.LastName,
        u.FirstName,

        CASE 
            WHEN uf.UserId IS NOT NULL THEN 1
            ELSE 0
        END AS IsFriend,

    FROM UserData u

CREATE PROCEDURE [dbo].[GetUserPosts]
    @UserId INT
AS
BEGIN
    SELECT 
        p.PostId,
        p.UserId,
        p.PostContent,
        p.PostPhoto,
        p.PostDate,
        p.LikeCount,
        p.ShareCount,
        p.CommentCount,
        ud.FirstName,
        ud.LastName,
        ud.ProfilePhoto,
        p.Status,
        CASE 
            WHEN pl.UserId IS NOT NULL THEN 1
            ELSE 0
        END AS IsLiked,
        pl.LikeType
    FROM UserPosts p
    INNER JOIN UserData ud ON p.UserId = ud.UserId
    LEFT JOIN PostLikes pl ON p.PostId = pl.PostId AND pl.UserId = @UserId
    WHERE p.Status IS NULL 
    AND (p.UserId = @UserId 
        OR EXISTS (
            SELECT 1 
            FROM UserFriends uf 
            WHERE (uf.UserId = @UserId AND uf.FollowerId = p.UserId AND uf.RequestStatus = 'accepted') 
            OR (uf.FollowerId = @UserId AND uf.UserId = p.UserId AND uf.RequestStatus = 'accepted')
        ))
    ORDER BY p.PostDate;
END

CREATE PROCEDURE [dbo].[LikePost]
    @PostId INT,
    @UserId INT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @IsLiked BIT;

    -- Check if the post exists
    IF EXISTS (SELECT 1 FROM UserPosts WHERE PostId = @PostId)
    BEGIN
        -- Check if the user has already liked the post
        IF NOT EXISTS (SELECT 1 FROM PostLikes WHERE PostId = @PostId AND UserId = @UserId)
        BEGIN
            -- User hasn't liked the post, so add a new like
            INSERT INTO PostLikes (PostId, UserId, LikeDate)
            VALUES (@PostId, @UserId, GETDATE());

            -- Increment like count of the post
            UPDATE UserPosts
            SET LikeCount = LikeCount + 1
            WHERE PostId = @PostId;

            SET @IsLiked = 1; -- User has liked the post
        END
        ELSE
        BEGIN
            -- User has already liked the post, so remove the like
            DELETE FROM PostLikes
            WHERE PostId = @PostId AND UserId = @UserId;

            -- Decrement like count of the post
            UPDATE UserPosts
            SET LikeCount = LikeCount - 1
            WHERE PostId = @PostId;

            SET @IsLiked = 0; -- User has unliked the post
        END

        -- Return updated like count and isLiked status
        SELECT LikeCount, @IsLiked AS isLiked
        FROM UserPosts
        WHERE PostId = @PostId;
    END
    ELSE
    BEGIN
        -- Post not found
        SELECT -1 AS LikeCount, CAST(NULL AS BIT) AS isLiked;
    END
END
GO

CREATE TRIGGER [dbo].[trg_PostCommentNotification]
ON [dbo].[PostComments]
AFTER INSERT
AS
BEGIN
    INSERT INTO Notifications (UserID, NotificationType, NotificationSenderUserID, NotificationPostID, NotificationText, ProfilePhoto, PostContent, PostPhoto)
    SELECT
        p.UserId AS UserID,  -- The user receiving the notification
        'Comment' AS NotificationType,
        pc.UserId AS NotificationSenderUserID,  -- The user who commented
        pc.PostId AS NotificationPostID,
        'User ' + u.FirstName + ' ' + u.LastName + ' commented on your post.' AS NotificationText,
        u.ProfilePhoto AS ProfilePhoto,
        p.PostContent AS PostContent,
        p.PostPhoto AS PostPhoto
    FROM inserted pc
    INNER JOIN UserPosts p ON pc.PostId = p.PostId
    INNER JOIN UserData u ON pc.UserId = u.UserId;
END;
